515 void vimChannelMap(uint32 request, uint32 channel, t_isrFuncPTR handler)
516 {
517  uint32 i,j;
518  i = channel >> 2U; /* Find the register to configure */
519  j = channel - (i << 2U); /* Find the offset of the type */
520  j = 3U - j; /* reverse the byte order */
521  j = j << 3U; /* find the bit location */
522 
523  /*Mapping the required interrupt request to the required channel*/
524  vimREG->CHANCTRL[i] &= ~(uint32)((uint32)0xFFU << j);
525  vimREG->CHANCTRL[i] |= (request << j);
526 
527  /*Updating VIMRAM*/
528  vimRAM->ISR[channel + 1U] = handler;
529 }
