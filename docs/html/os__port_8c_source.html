150 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters, BaseType_t xRunPrivileged )
151 {
152 StackType_t *pxOriginalTOS;
153 
154  pxOriginalTOS = pxTopOfStack;
155 
156  /* Ensure the stack is correctly aligned on exit. */
157  pxTopOfStack--;
158 
159  /* Setup the initial stack of the task. The stack is set exactly as
160  expected by the portRESTORE_CONTEXT() macro. */
161 
162  /* First on the stack is the return address - which is the start of the as
163  the task has not executed yet. The offset is added to make the return
164  address appear as it would within an IRQ ISR. */
165  *pxTopOfStack = ( StackType_t ) pxCode + portINSTRUCTION_SIZE;
166  pxTopOfStack--;
167 
168  *pxTopOfStack = ( StackType_t ) 0x00000000; /* R14 */
169  pxTopOfStack--;
170  *pxTopOfStack = ( StackType_t ) pxOriginalTOS; /* Stack used when task starts goes in R13. */
171  pxTopOfStack--;
172 
173  #ifdef portPRELOAD_TASK_REGISTERS
174  {
175  *pxTopOfStack = ( StackType_t ) 0x12121212; /* R12 */
176  pxTopOfStack--;
177  *pxTopOfStack = ( StackType_t ) 0x11111111; /* R11 */
178  pxTopOfStack--;
179  *pxTopOfStack = ( StackType_t ) 0x10101010; /* R10 */
180  pxTopOfStack--;
181  *pxTopOfStack = ( StackType_t ) 0x09090909; /* R9 */
182  pxTopOfStack--;
183  *pxTopOfStack = ( StackType_t ) 0x08080808; /* R8 */
184  pxTopOfStack--;
185  *pxTopOfStack = ( StackType_t ) 0x07070707; /* R7 */
186  pxTopOfStack--;
187  *pxTopOfStack = ( StackType_t ) 0x06060606; /* R6 */
188  pxTopOfStack--;
189  *pxTopOfStack = ( StackType_t ) 0x05050505; /* R5 */
190  pxTopOfStack--;
191  *pxTopOfStack = ( StackType_t ) 0x04040404; /* R4 */
192  pxTopOfStack--;
193  *pxTopOfStack = ( StackType_t ) 0x03030303; /* R3 */
194  pxTopOfStack--;
195  *pxTopOfStack = ( StackType_t ) 0x02020202; /* R2 */
196  pxTopOfStack--;
197  *pxTopOfStack = ( StackType_t ) 0x01010101; /* R1 */
198  pxTopOfStack--;
199  }
200  #else
201  {
202  pxTopOfStack -= portSPACE_BETWEEN_TOS_AND_PARAMETERS;
203  }
204  #endif
205 
206  /* Function parameters are passed in R0. */
207  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
208  pxTopOfStack--;
209 
210  /* Set the status register for system or user mode, with interrupts enabled. */
211  if( xRunPrivileged == pdTRUE )
212  {
213  *pxTopOfStack = (portSTACK_TYPE) ((_getCPSRValue_() & ~0xFF) | portINITIAL_SPSR_IF_PRIVILEGED);
214  }
215  else
216  {
217  *pxTopOfStack = (portSTACK_TYPE) ((_getCPSRValue_() & ~0xFF) | portINITIAL_SPSR_IF_UNPRIVILEGED);
218  }
219 
220  if( ( ( uint32_t ) pxCode & 0x01UL ) != 0x00 )
221  {
222  /* The task will start in thumb mode. */
223  *pxTopOfStack |= portTHUMB_MODE_BIT;
224  }
225  pxTopOfStack--;
226 
227  /* The last thing on the stack is the tasks ulUsingFPU value, which by
228  default is set to indicate that the stack frame does not include FPU
229  registers. */
230  *pxTopOfStack = pdFALSE;
231 
232  return pxTopOfStack;
233 }

237 static unsigned long prvGetMPURegionSizeSetting( unsigned long ulActualSizeInBytes )
238 {
239  unsigned long ulRegionSize, ulReturnValue = 4;
240 
241  /* 32 is the smallest region size, 31 is the largest valid value for
242  ulReturnValue. */
243  for( ulRegionSize = 32UL; ulReturnValue < 31UL; ( ulRegionSize <<= 1UL ) )
244  {
245  if( ulActualSizeInBytes <= ulRegionSize )
246  {
247  break;
248  }
249  else
250  {
251  ulReturnValue++;
252  }
253  }
254 
255  /* Shift the code by one before returning so it can be written directly
256  into the the correct bit position of the attribute register. */
257  return ulReturnValue << 1UL;
258 }

262 void vPortStoreTaskMPUSettings( xMPU_SETTINGS *xMPUSettings, const struct xMEMORY_REGION * const xRegions, StackType_t *pxBottomOfStack, uint32_t usStackDepth )
263 {
264  long lIndex;
265  unsigned long ul;
266 
267  if( xRegions == NULL )
268  {
269 /* USER CODE BEGIN (6) */
270 /* USER CODE END */
271  /* No MPU regions are specified so allow access to all of the RAM. */
272  xMPUSettings->xRegion[0].ulRegionBaseAddress = 0x08000000;
273  xMPUSettings->xRegion[0].ulRegionSize = portMPU_SIZE_512KB | portMPU_REGION_ENABLE;
274  xMPUSettings->xRegion[0].ulRegionAttribute = portMPU_PRIV_RW_USER_RW_EXEC | portMPU_NORMAL_OIWTNOWA_SHARED;
275 
276  /* Re-instate the privileged only RAM region as xRegion[ 0 ] will have
277  just removed the privileged only parameters. */
278  xMPUSettings->xRegion[1].ulRegionBaseAddress = 0x08000000;
279  xMPUSettings->xRegion[1].ulRegionSize = portMPU_SIZE_4KB | portMPU_REGION_ENABLE;
280  xMPUSettings->xRegion[1].ulRegionAttribute = portMPU_PRIV_RW_USER_NA_NOEXEC | portMPU_NORMAL_OIWTNOWA_SHARED;
281 
282 /* USER CODE BEGIN (7) */
283 /* USER CODE END */
284  /* Invalidate all other regions. */
285  for( ul = 2; ul <= portNUM_CONFIGURABLE_REGIONS; ul++ )
286  {
287  xMPUSettings->xRegion[ ul ].ulRegionBaseAddress = 0x00000000UL;
288  xMPUSettings->xRegion[ ul ].ulRegionSize = 0UL;
289  xMPUSettings->xRegion[ ul ].ulRegionAttribute = 0UL;
290  }
291 /* USER CODE BEGIN (8) */
292 /* USER CODE END */
293  }
294  else
295  {
296 /* USER CODE BEGIN (9) */
297 /* USER CODE END */
298  /* This function is called automatically when the task is created - in
299  which case the stack region parameters will be valid. At all other
300  times the stack parameters will not be valid and it is assumed that the
301  stack region has already been configured. */
302  if( usStackDepth > 0 )
303  {
304  /* Define the region that allows access to the stack. */
305  xMPUSettings->xRegion[0].ulRegionBaseAddress = (unsigned)pxBottomOfStack;
306  xMPUSettings->xRegion[0].ulRegionSize = prvGetMPURegionSizeSetting( (unsigned long)usStackDepth * (unsigned long) sizeof(portSTACK_TYPE) ) | portMPU_REGION_ENABLE;
307  xMPUSettings->xRegion[0].ulRegionAttribute = portMPU_REGION_READ_WRITE | portMPU_NORMAL_OIWTNOWA_SHARED;
308 
309  }
310 /* USER CODE BEGIN (10) */
311 /* USER CODE END */
312  lIndex = 0;
313 
314  for( ul = 1; ul <= portNUM_CONFIGURABLE_REGIONS; ul++ )
315  {
316  if( ( xRegions[ lIndex ] ).ulLengthInBytes > 0UL )
317  {
318  /* Translate the generic region definition contained in
319  xRegions into the R4 specific MPU settings that are then
320  stored in xMPUSettings. */
321  xMPUSettings->xRegion[ul].ulRegionBaseAddress = (unsigned long) xRegions[lIndex].pvBaseAddress;
322  xMPUSettings->xRegion[ul].ulRegionSize = prvGetMPURegionSizeSetting( xRegions[ lIndex ].ulLengthInBytes ) | portMPU_REGION_ENABLE;
323  xMPUSettings->xRegion[ul].ulRegionAttribute = xRegions[ lIndex ].ulParameters;
324  }
325  else
326  {
327  /* Invalidate the region. */
328  xMPUSettings->xRegion[ ul ].ulRegionBaseAddress = 0x00000000UL;
329  xMPUSettings->xRegion[ ul ].ulRegionSize = 0UL;
330  xMPUSettings->xRegion[ ul ].ulRegionAttribute = 0UL;
331  }
332  lIndex++;
333  }
334 /* USER CODE BEGIN (11) */
335 /* USER CODE END */
336  }

338 /* USER CODE END */

344 {
345  /* make sure MPU is disabled */
346  prvMpuDisable();
347 
348  /* First setup the entire flash for unprivileged read only access. */
349  prvMpuSetRegion(portUNPRIVILEGED_FLASH_REGION, 0x00000000, portMPU_SIZE_4MB | portMPU_REGION_ENABLE, portMPU_PRIV_RO_USER_RO_EXEC | portMPU_NORMAL_OIWTNOWA_SHARED);
350 
351  /* Setup the first 32K for privileged only access. This is where the kernel code is
352  placed. */
353  prvMpuSetRegion(portPRIVILEGED_FLASH_REGION, 0x00000000, portMPU_SIZE_32KB | portMPU_REGION_ENABLE, portMPU_PRIV_RO_USER_NA_EXEC | portMPU_NORMAL_OIWTNOWA_SHARED);
354 
355  /* Setup the the entire RAM region for privileged read-write and unprivileged read only access */
356  prvMpuSetRegion(portPRIVILEGED_RAM_REGION, 0x08000000, portMPU_SIZE_512KB | portMPU_REGION_ENABLE, portMPU_PRIV_RW_USER_RO_EXEC | portMPU_NORMAL_OIWTNOWA_SHARED);
357 
358  /* Default peripherals setup */
359  prvMpuSetRegion(portGENERAL_PERIPHERALS_REGION, 0xF0000000,

366  /* Enable MPU */
367  prvMpuEnable();

373 static void prvSetupTimerInterrupt(void)
374 {
375  /* Disable timer 0. */
376  portRTI_GCTRL_REG &= 0xFFFFFFFEUL;
377 
378  /* Use the internal counter. */
379  portRTI_TBCTRL_REG = 0x00000000U;
380 
381  /* COMPSEL0 will use the RTIFRC0 counter. */
382  portRTI_COMPCTRL_REG = 0x00000000U;
383 
384  /* Initialise the counter and the prescale counter registers. */
385  portRTI_CNT0_UC0_REG = 0x00000000U;
386  portRTI_CNT0_FRC0_REG = 0x00000000U;
387 
388  /* Set Prescalar for RTI clock. */
389  portRTI_CNT0_CPUC0_REG = 0x00000001U;
390  portRTI_CNT0_COMP0_REG = ( configCPU_CLOCK_HZ / 2 ) / configTICK_RATE_HZ;
391  portRTI_CNT0_UDCP0_REG = ( configCPU_CLOCK_HZ / 2 ) / configTICK_RATE_HZ;
392 
393  /* Clear interrupts. */
394  portRTI_INTFLAG_REG = 0x0007000FU;
395  portRTI_CLEARINTENA_REG = 0x00070F0FU;
396 
397  /* Enable the compare 0 interrupt. */
398  portRTI_SETINTENA_REG = 0x00000001U;
399  portRTI_GCTRL_REG |= 0x00000001U;
400 }

421  /* Should not get here! */
