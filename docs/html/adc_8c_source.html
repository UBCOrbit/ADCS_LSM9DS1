627 void adcCalibration(adcBASE_t *adc)
628 {
629 /* USER CODE BEGIN (25) */
630 /* USER CODE END */
631 
632  uint32 conv_val[5U]={0U,0U,0U,0U,0U};
633  uint32 loop_index=0U;
634  uint32 offset_error=0U;
635  uint32 backup_mode;
636 
638  backup_mode = adc->OPMODECR;
639 
641  adc->OPMODECR |= 0x80000000U;
642 
643  /* Disable all channels for conversion */
644  adc->GxSEL[0U]=0x00U;
645  adc->GxSEL[1U]=0x00U;
646  adc->GxSEL[2U]=0x00U;
647 
648  for(loop_index=0U;loop_index<4U;loop_index++)
649  {
650  /* Disable Self Test and Calibration mode */
651  adc->CALCR=0x0U;
652 
653  switch(loop_index)
654  {
655  case 0U : /* Test 1 : Bride En = 0 , HiLo =0 */
656  adc->CALCR=0x0U;
657  break;
658 
659  case 1U : /* Test 1 : Bride En = 0 , HiLo =1 */
660  adc->CALCR=0x0100U;
661  break;
662 
663  case 2U : /* Test 1 : Bride En = 1 , HiLo =0 */
664  adc->CALCR=0x0200U;
665  break;
666 
667  case 3U : /* Test 1 : Bride En = 1 , HiLo =1 */
668  adc->CALCR=0x0300U;
669  break;
670  default :
671  break;
672  }
673 
674  /* Enable Calibration mode */
675  adc->CALCR|=0x1U;
676 
677  /* Start calibration conversion */
678  adc->CALCR|=0x00010000U;
679 
680  /* Wait for calibration conversion to complete */
681  /*SAFETYMCUSW 28 D MR:NA <APPROVED> "Hardware status bit read check" */
682  while((adc->CALCR & 0x00010000U)==0x00010000U)
683  {
684  } /* Wait */
685 
686  /* Read converted value */
687  conv_val[loop_index]= adc->CALR;
688  }
689 
690  /* Disable Self Test and Calibration mode */
691  adc->CALCR=0x0U;
692 
693  /* Compute the Offset error correction value */
694  conv_val[4U]=conv_val[0U]+ conv_val[1U] + conv_val[2U] + conv_val[3U];
695 
696  conv_val[4U]=(conv_val[4U]/4U);
697 
698  offset_error=conv_val[4U]-0x7FFU;
699 
700  /*Write the offset error to the Calibration register */
701  /* Load 2;s complement of the computed value to ADCALR register */
702  offset_error=~offset_error;
703  offset_error=offset_error & 0xFFFU;
704  offset_error=offset_error+1U;
705 
706  adc->CALR = offset_error;
707 
709  adc->OPMODECR = backup_mode;
710 
713 /* USER CODE BEGIN (26) */
714 /* USER CODE END */
715 }
