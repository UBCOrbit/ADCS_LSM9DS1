92 void dmaReqAssign(uint32 channel,uint32 reqline)
93 {
94  register uint32 i=0U,j=0U;
95 
96  i = channel >> 2U; /* Find the register to configure */
97  j = channel - (i << 2U); /* Find the offset of the type */
98  j = 3U - j; /* reverse the byte order */
99  j = j << 3U; /* find the bit location */
100 
101  /* mapping channel 'i' to request line 'j' */
102  dmaREG->DREQASI[i] &= ~(uint32)((uint32)0xFFU << j);
103  dmaREG->DREQASI[i] |= (reqline << j);
104 }

136 void dmaSetCtrlPacket(uint32 channel, g_dmaCTRL g_dmaCTRLPKT)
137 {
138  register uint32 i=0U,j=0U;
139 
140  dmaRAMREG->PCP[channel].ISADDR = g_dmaCTRLPKT.SADD;
141 
142  dmaRAMREG->PCP[channel].IDADDR = g_dmaCTRLPKT.DADD;
143 
144  dmaRAMREG->PCP[channel].ITCOUNT = (g_dmaCTRLPKT.FRCNT << 16U) | g_dmaCTRLPKT.ELCNT;
145 
146  dmaRAMREG->PCP[channel].CHCTRL = (g_dmaCTRLPKT.RDSIZE << 14U) | (g_dmaCTRLPKT.WRSIZE << 12U) | (g_dmaCTRLPKT.TTYPE << 8U)| \
147  (g_dmaCTRLPKT.ADDMODERD << 3U ) | (g_dmaCTRLPKT.ADDMODEWR << 1U ) | (g_dmaCTRLPKT.AUTOINIT);
148 
149  dmaRAMREG->PCP[channel].CHCTRL |= (g_dmaCTRLPKT.CHCTRL << 16U);
150 
151  dmaRAMREG->PCP[channel].EIOFF = (g_dmaCTRLPKT.ELDOFFSET << 16U) | (g_dmaCTRLPKT.ELSOFFSET);
152 
153  dmaRAMREG->PCP[channel].FIOFF = (g_dmaCTRLPKT.FRDOFFSET << 16U) | (g_dmaCTRLPKT.FRSOFFSET);
154 
155  i = channel >> 3U; /* Find the register to write */
156  j = channel - (i << 3U); /* Find the offset of the 4th bit */
157  j = 7U - j; /* Reverse the order of the 4th bit offset */
158  j = j << 2U; /* Find the bit location of the 4th bit to write */
159 
160  dmaREG->PAR[i] &= ~(uint32)((uint32)0xFU << j);
161  dmaREG->PAR[i] |= (g_dmaCTRLPKT.PORTASGN << j);
162 }
